# PURPOSE:

#   outdated dependencies create supply-chain risk (OWASP A03:2025 — Software

# DEMO COMMAND:
#   pip3 install pip-audit --break-system-packages
#   pip-audit -r requirements_vulnerable.txt

# Flask 1.0.0  (released April 2018)
#   CVE-2023-30861 — cookie handling bug allows session persistence after logout
#   CVE-2018-1000656 — possible denial of service via large request
Flask==1.0.0

# Werkzeug 0.14.1  (released January 2018 — the WSGI library Flask is built on)
#   CVE-2023-25577 — multipart form parsing causes excessive resource consumption
#   CVE-2022-29361 — URL parsing bypass allows host header injection
#   CVE-2019-14806 — debug PIN predictable in certain configurations (→ RCE)
Werkzeug==0.14.1

# Jinja2 2.10  (released November 2017 — the templating engine used by Flask)
#   CVE-2019-10906 — sandbox escape via _fail_with_undefined_error
#   CVE-2019-8341 — server-side template injection possible in certain contexts
Jinja2==2.10

# requests 2.18.0  (released June 2017 — HTTP library)
#   CVE-2018-18074 — HTTPS credentials leaked to HTTP redirect targets
#   CVE-2023-32681 — Proxy-Authorization header forwarded to destination server
requests==2.18.0

# cryptography 2.1.4  (released November 2017)
#   Multiple CVEs relating to RSA, DSA, and symmetric cipher implementations
#   CVE-2020-25659, CVE-2020-36242, CVE-2023-0286, CVE-2023-23931
cryptography==2.1.4

# SQLAlchemy 1.2.0  (released January 2018 — ORM used in many Flask apps)
#   CVE-2019-7548 — SQL injection possible in certain ORDER BY clauses
SQLAlchemy==1.2.0

# itsdangerous 0.24  (released June 2014 — used by Flask for session signing)
#   Predates several improvements to timing-safe comparison operations
#   Required by Flask 1.0.0 at this version — shows transitive dependency risk
itsdangerous==0.24
